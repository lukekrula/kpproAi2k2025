    @GetMapping("/detail/{id}")
    public String detail(Model model, @PathVariable int id) {
        Item item = itemService.get(id);
        if(item != null) {
            model.addAttribute("item", item);
            return "detail";
        }else{
            return "redirect:/items/";
        }
    }
